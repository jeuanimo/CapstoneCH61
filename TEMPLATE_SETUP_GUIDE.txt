BASE TEMPLATE (base.html) - Use for all pages
================================================================================

{% load static %}
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>{% block title %}Capstone CH61{% endblock %}</title>
    
    <!-- Global CSS -->
    <link rel="stylesheet" href="{% static 'css/base.css' %}">
    <link rel="stylesheet" href="{% static 'css/navbar.css' %}">
    
    <!-- Page-specific CSS - Override in child templates -->
    {% block extra_css %}{% endblock %}
</head>
<body>
    <!-- Top Bar with Brand and Social Links -->
    <div class="top-bar">
        <div class="container">
            <div class="brand-row">
                <div class="brand-wrapper">
                    <a href="{% url 'home' %}" class="navbar-brand">
                        <div class="brand-name">YourName</div>
                        <div class="brand-subtitle">Portfolio</div>
                    </a>
                </div>
                <div class="top-actions">
                    <div class="social-media">
                        <!-- Add social links here -->
                        <a href="#" class="social-link" title="GitHub">
                            <i class="fab fa-github"></i>
                        </a>
                        <a href="#" class="social-link" title="LinkedIn">
                            <i class="fab fa-linkedin"></i>
                        </a>
                        <a href="#" class="social-link" title="Email">
                            <i class="fas fa-envelope"></i>
                        </a>
                    </div>
                    {% if user.is_authenticated %}
                    <div class="user-badge">
                        <i class="fas fa-user-circle"></i>
                        <span class="username">{{ user.username }}</span>
                    </div>
                    {% endif %}
                    <button class="theme-toggle-btn" id="themeToggle">
                        <i class="fas fa-moon"></i>
                        <span class="theme-label">Dark</span>
                    </button>
                </div>
            </div>
        </div>
    </div>

    <!-- Main Navigation -->
    <nav class="navbar">
        <div class="container">
            <div class="navbar-collapse">
                <ul class="navbar-nav">
                    <li class="nav-item {% if request.resolver_match.url_name == 'home' %}active{% endif %}">
                        <a href="{% url 'home' %}" class="nav-link">Home</a>
                    </li>
                    <li class="nav-item {% if request.resolver_match.url_name == 'about' %}active{% endif %}">
                        <a href="{% url 'about' %}" class="nav-link">About</a>
                    </li>
                    <li class="nav-item {% if request.resolver_match.url_name == 'projects' %}active{% endif %}">
                        <a href="{% url 'projects' %}" class="nav-link">Projects</a>
                    </li>
                    <li class="nav-item {% if request.resolver_match.url_name == 'contact' %}active{% endif %}">
                        <a href="{% url 'contact' %}" class="nav-link">Contact</a>
                    </li>
                </ul>
            </div>
        </div>
    </nav>

    <!-- Messages/Alerts -->
    {% if messages %}
    <div class="container mt-2">
        {% for message in messages %}
        <div class="alert alert-{{ message.tags }}">
            {{ message }}
        </div>
        {% endfor %}
    </div>
    {% endif %}

    <!-- Main Content -->
    <main class="main-content">
        <div class="container">
            {% block content %}{% endblock %}
        </div>
    </main>

    <!-- Footer -->
    <footer>
        <p>&copy; 2026 Capstone CH61. All rights reserved.</p>
    </footer>

    <!-- JavaScript for theme toggle -->
    <script>
        document.getElementById('themeToggle').addEventListener('click', function() {
            document.body.classList.toggle('dark-mode');
            localStorage.setItem('theme', document.body.classList.contains('dark-mode') ? 'dark' : 'light');
        });
        
        // Load saved theme preference
        if (localStorage.getItem('theme') === 'dark') {
            document.body.classList.add('dark-mode');
        }
    </script>
</body>
</html>

================================================================================
NOTES
================================================================================

1. Replace URL names with your actual URL names from urls.py
2. Customize brand name, social links, and content as needed
3. Add Font Awesome or icon library to base.html head:
   <link rel="stylesheet" href="https://cdnjs.cloudflare.com/ajax/libs/font-awesome/6.0.0/css/all.min.css">
4. All forms automatically get dark mode support
5. Navigation highlights current page automatically
6. Theme toggle saves preference to localStorage
